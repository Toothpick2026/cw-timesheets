-- ===================================================================== -- CorpWorks Time — Stage 2, Part 2: let owner adjust actual times -- Replaces owner_adjust_timesheet so it can also correct the -- START time, FINISH time and LUNCH minutes (not just paid hours). -- SAFE TO RUN ON LIVE DB. Only touches this one function. -- ===================================================================== -- Remove the old 3-argument version so there is no ambiguity. drop function if exists public.owner_adjust_timesheet(bigint, numeric, text); -- New version: id, start, finish, lunch minutes, paid hours, payroll note. create or replace function public.owner_adjust_timesheet( p_id bigint, p_start_time text, p_end_time text, p_lunch_break int, p_paid_hours numeric, p_payroll_note text default null ) returns void language plpgsql security definer set search_path to 'public', 'extensions' as $$ declare v_org uuid; begin v_org := public.current_owner_org(); if v_org is null then raise exception 'Not an owner account.'; end if; if p_paid_hours is null or p_paid_hours < 0 or p_paid_hours > 24 then raise exception 'Paid hours must be between 0 and 24.'; end if; if p_lunch_break is null or p_lunch_break < 0 or p_lunch_break > 600 then raise exception 'Lunch minutes must be between 0 and 600.'; end if; update public.timesheets t set start_time = p_start_time, end_time = p_end_time, lunch_break = p_lunch_break, paid_hours = p_paid_hours, payroll_note = p_payroll_note where t.id = p_id and exists (select 1 from public.staff s where s.name = t.staff and s.org_id = v_org); if not found then raise exception 'Timesheet not found for your organisation.'; end if; end; $$; revoke all on function public.owner_adjust_timesheet(bigint, text, text, int, numeric, text) from public; grant execute on function public.owner_adjust_timesheet(bigint, text, text, int, numeric, text) to authenticated;
